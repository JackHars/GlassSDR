## Background: POCSAG in 60 seconds

- **POCSAG** (Post Office Code Standardisation Advisory Group): paging protocol, 3 baud rates: 512, 1200, 2400 bps.
- **Frame structure**: 576-bit preamble (alternating 10101...) → one or more batches. Each batch: 1 sync codeword (0x7CD215D8) + 8 frames (2 codewords each = 16 codewords per batch).
- **Codeword**: 32 bits = 1 flag bit + 20 data bits + 10 BCH(31,21) parity bits + 1 even-parity bit.
  - Address codeword: flag=0, 18-bit address (RIC bits 2-19) + 2 function bits.
  - Message codeword: flag=1, 20 message bits.
  - Idle codeword: `0x7A89C197` (fills unused frame slots).
- **RIC (Radio Identity Code)**: 21 bits total. Bits 0-2 encode which of the 8 frame positions the address appears in. Bits 2-20 go into the address codeword.
- **Numeric encoding**: 4-bit BCD per character (0-9, *, U, space, -, ], [).
- **Alphanumeric encoding**: 7-bit ASCII packed into 20-bit message codeword chunks.
- **BCH(31,21)**: Generator polynomial 0x769 (x^10 + x^9 + x^8 + x^6 + x^5 + x^3 + 1). Systematic encoding: shift data left 10, divide by generator, XOR remainder.
- **2-FSK modulation**: ±4.5 kHz deviation at the given baud rate. Gaussian filtering (BT=0.5) smooths transitions.

```rust
//! BCH(31,21) encoder for POCSAG codewords.
//!
//! Generator polynomial: g(x) = x^10 + x^9 + x^8 + x^6 + x^5 + x^3 + 1
//! In binary (11 bits, degree 10): 0b111_0110_1001 = 0x769

const GENERATOR: u32 = 0x769;

/// Compute BCH(31,21) parity bits for 21 data bits.
/// Returns a 10-bit parity value.
///
/// The encoding is systematic: the transmitted 31-bit word is
/// `[21 data bits][10 parity bits]`, where parity = remainder of
/// (data << 10) / generator.
pub fn encode_bch(data: u32) -> u32 {
    debug_assert!(data < (1 << 21), "data must be 21 bits");
    let mut remainder = data << 10; // 31-bit shifted value
    for i in (10..31).rev() {
        if remainder & (1 << i) != 0 {
            remainder ^= GENERATOR << (i - 10);
        }
    }
    remainder & 0x3FF // bottom 10 bits
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn zero_data_gives_zero_parity() {
        assert_eq!(encode_bch(0), 0);
    }

    #[test]
    fn parity_is_10_bits() {
        // Exhaustive check is too slow; spot-check boundaries
        for d in [1u32, 0x1FFFFF, 0x100000, 0x0AAAAA] {
            assert!(encode_bch(d) < 1024);
        }
    }

    #[test]
    fn known_idle_codeword() {
        // Idle codeword is 0x7A89C197.
        // Structure: flag=0, 20 data bits, 10 BCH bits, 1 even-parity bit.
        // Bits 31..11 (21 bits): 0_11110101000100111000 = 0x1EA270 ... let's extract properly.
        // 0x7A89C197 in binary:
        //   0111 1010 1000 1001 1100 0001 1001 0111
        // bit31=0 (flag), bits30..11 = 1111010100010011100000 (20 data bits),
        //   wait — flag is bit 31. data is bits 30..11.
        // Actually the POCSAG codeword format is:
        //   bit 0 (MSB transmitted first) = flag
        //   bits 1-20 = data (address or message)
        //   bits 21-30 = BCH parity
        //   bit 31 = even parity over bits 0-30
        // So for idle 0x7A89C197:
        //   As 32-bit big-endian: 0x7A89C197
        //   flag (bit 31) = 0
        //   data (bits 30-11) = (0x7A89C197 >> 11) & 0xFFFFF = 0xF5138
        //   BCH (bits 10-1) = (0x7A89C197 >> 1) & 0x3FF = 0x0CB
        //   parity (bit 0) = 1
        let idle_data: u32 = (0x7A89C197u32 >> 11) & 0xFFFFF;
        // For BCH, the 21-bit input is flag + data = 0 concatenated with idle_data
        // Actually: 21 bits = 1 flag bit + 20 data bits
        let twenty_one = (0u32 << 20) | idle_data; // flag=0
        let expected_bch = (0x7A89C197u32 >> 1) & 0x3FF;
        assert_eq!(encode_bch(twenty_one), expected_bch);
    }
}
```

```rust
//! POCSAG codeword construction.
//!
//! Codeword format (32 bits, MSB-first):
//!   bit 31: flag (0 = address, 1 = message)
//!   bits 30-11: 20 data bits
//!   bits 10-1: BCH(31,21) parity
//!   bit 0: even parity over bits 31..1

use super::bch::encode_bch;

/// POCSAG sync codeword (constant, not BCH-protected).
pub const fn sync_codeword() -> u32 {
    0x7CD215D8
}

/// POCSAG idle codeword (constant, valid BCH + even parity).
pub const fn idle_codeword() -> u32 {
    0x7A89C197
}

/// Build an address codeword.
///
/// - `ric`: Radio Identity Code (0..=2_097_151, 21 bits). Bits 0-2 determine the
///   frame position within a batch; bits 2-20 are the 18-bit address field.
/// - `function`: function bits (0..=3).
///
/// Returns the fully assembled 32-bit codeword.
pub fn address_codeword(ric: u32, function: u8) -> u32 {
    debug_assert!(ric < (1 << 21));
    debug_assert!(function < 4);

    let addr_field = (ric >> 3) & 0x3FFFF; // 18 bits
    let func_field = (function as u32) & 0x03;

    // 20 data bits = 18 address + 2 function
    let data_20 = (addr_field << 2) | func_field;
    assemble_codeword(0, data_20)
}

/// Build a message codeword from 20 packed message bits.
pub fn message_codeword(data_20: u32) -> u32 {
    debug_assert!(data_20 < (1 << 20));
    assemble_codeword(1, data_20)
}

/// Assemble a 32-bit codeword from flag + 20 data bits.
fn assemble_codeword(flag: u32, data_20: u32) -> u32 {
    // 21-bit BCH input = flag (1 bit) + data (20 bits)
    let bch_input = (flag << 20) | data_20;
    let parity_10 = encode_bch(bch_input);

    // 31 bits before even parity: [flag:1][data:20][bch:10]
    let word_31 = (flag << 30) | (data_20 << 10) | parity_10;

    // Even parity bit (bit 0): makes total popcount even
    let even_parity = (word_31.count_ones() % 2) as u32;
    (word_31 << 1) | even_parity
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn idle_codeword_is_valid() {
        let idle = idle_codeword();
        // Even parity check
        assert_eq!(idle.count_ones() % 2, 0);
    }

    #[test]
    fn assemble_then_verify_parity() {
        for data in [0u32, 0xFFFFF, 0x55555, 0xAAAAA] {
            let cw = assemble_codeword(0, data);
            assert_eq!(cw.count_ones() % 2, 0);
            let cw = assemble_codeword(1, data);
            assert_eq!(cw.count_ones() % 2, 0);
        }
    }
}
```

```rust
//! POCSAG character encoding: numeric (BCD, 4-bit) and alphanumeric (7-bit ASCII LSB-first).

/// Encode a numeric string into a bit vector (4 bits per character, MSB first).
///
/// Valid characters: 0-9, *, U, space, -, ], [
/// Invalid characters are replaced with space (0xC).
pub fn encode_numeric(msg: &str) -> Vec<bool> {
    let mut bits = Vec::with_capacity(msg.len() * 4);
    for ch in msg.chars() {
        let nibble = match ch {
            '0' => 0x0u8,
            '1' => 0x1,
            '2' => 0x2,
            '3' => 0x3,
            '4' => 0x4,
            '5' => 0x5,
            '6' => 0x6,
            '7' => 0x7,
            '8' => 0x8,
            '9' => 0x9,
            '*' => 0xA,
            'U' | 'u' => 0xB,
            ' ' => 0xC,
            '-' => 0xD,
            ']' => 0xE,
            '[' => 0xF,
            _ => 0xC, // default to space
        };
        // 4 bits, MSB first
        for i in (0..4).rev() {
            bits.push((nibble >> i) & 1 == 1);
        }
    }
    bits
}

/// Encode an alphanumeric string into a bit vector (7 bits per character, LSB first).
///
/// Characters are 7-bit ASCII. Non-ASCII or values >= 128 are replaced with '?'.
pub fn encode_alphanumeric(msg: &str) -> Vec<bool> {
    let mut bits = Vec::with_capacity(msg.len() * 7);
    for ch in msg.chars() {
        let val = if ch.is_ascii() && (ch as u32) < 128 {
            ch as u8
        } else {
            b'?'
        };
        // 7 bits, LSB first (POCSAG convention)
        for i in 0..7 {
            bits.push((val >> i) & 1 == 1);
        }
    }
    bits
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn numeric_roundtrip() {
        let input = "0123456789";
        let bits = encode_numeric(input);
        assert_eq!(bits.len(), 40);
    }

    #[test]
    fn alpha_length() {
        let bits = encode_alphanumeric("Hello");
        assert_eq!(bits.len(), 35); // 5 * 7
    }
}
```

```rust
//! Full POCSAG message encoder: text → complete bitstream.
//!
//! Produces: 576-bit preamble + one or more batches (sync + 16 codewords each).

use super::charset::{encode_alphanumeric, encode_numeric};
use super::codeword::{address_codeword, idle_codeword, message_codeword, sync_codeword};

/// Message type determines encoding scheme.
#[derive(Debug, Clone)]
pub enum MessageType {
    Numeric(String),
    Alphanumeric(String),
    /// Tone-only (no message content, just the address).
    ToneOnly,
}

/// A complete POCSAG message to encode.
#[derive(Debug, Clone)]
pub struct PocsagMessage {
    pub ric: u32,
    pub function: u8,
    pub content: MessageType,
    pub baud_rate: u16, // 512, 1200, or 2400
}

/// Encode a POCSAG message into a complete bitstream (Vec<bool>).
///
/// The bitstream is ready for 2-FSK modulation: true = +deviation, false = -deviation.
pub fn encode_pocsag(msg: &PocsagMessage) -> Vec<bool> {
    let frame_pos = (msg.ric % 8) as usize;

    // Encode message content into bits, then pack into 20-bit message codeword chunks
    let content_bits = match &msg.content {
        MessageType::Numeric(s) => encode_numeric(s),
        MessageType::Alphanumeric(s) => encode_alphanumeric(s),
        MessageType::ToneOnly => vec![],
    };

    let msg_codewords = pack_message_codewords(&content_bits);

    // Build batch(es)
    // A batch has 8 frame slots (2 codewords each = 16 total).
    // The address goes at frame_pos. Message codewords follow immediately.
    // If the message spills past frame 7, we need another batch.
    let mut codewords_to_place: Vec<u32> = Vec::new();
    codewords_to_place.push(address_codeword(msg.ric, msg.function));
    codewords_to_place.extend(msg_codewords);

    let batches = build_batches(frame_pos, &codewords_to_place);

    // Assemble final bitstream
    let mut bits = Vec::new();

    // 576-bit preamble: alternating 1,0,1,0... (starting with 1)
    for i in 0..576 {
        bits.push(i % 2 == 0);
    }

    // Batches
    for batch in &batches {
        push_u32(&mut bits, sync_codeword());
        for &cw in batch {
            push_u32(&mut bits, cw);
        }
    }

    bits
}

/// Pack message bits into 20-bit chunks, padding the last chunk with zeros.
fn pack_message_codewords(content_bits: &[bool]) -> Vec<u32> {
    if content_bits.is_empty() {
        return vec![];
    }
    let mut codewords = Vec::new();
    for chunk in content_bits.chunks(20) {
        let mut data_20 = 0u32;
        for (i, &b) in chunk.iter().enumerate() {
            if b {
                data_20 |= 1 << (19 - i);
            }
        }
        codewords.push(message_codeword(data_20));
    }
    codewords
}

/// Build batches from a frame position and a list of codewords to place.
/// Returns Vec of 16-codeword arrays (each array is one batch).
fn build_batches(frame_pos: usize, codewords: &[u32]) -> Vec<[u32; 16]> {
    let mut batches = Vec::new();
    let mut batch = [idle_codeword(); 16];
    let mut slot = frame_pos * 2; // starting slot within the batch
    let mut cw_idx = 0;

    while cw_idx < codewords.len() {
        if slot >= 16 {
            // Current batch is full, start a new one
            batches.push(batch);
            batch = [idle_codeword(); 16];
            slot = 0;
        }
        batch[slot] = codewords[cw_idx];
        slot += 1;
        cw_idx += 1;
    }

    batches.push(batch);
    batches
}

/// Push a u32 as 32 bits (MSB first) into a bit vector.
fn push_u32(bits: &mut Vec<bool>, val: u32) {
    for i in (0..32).rev() {
        bits.push((val >> i) & 1 == 1);
    }
}
```

```rust
//! Gaussian-filtered 2-FSK modulator.
//!
//! Input: bool symbols (true = mark/+deviation, false = space/-deviation).
//! Output: Complex<f32> IQ samples at the specified sample rate.
//!
//! Parameters:
//! - `deviation_hz`: FSK deviation (±). POCSAG uses ±4500 Hz.
//! - `symbol_rate`: baud rate (512, 1200, or 2400 for POCSAG).
//! - `sample_rate`: output sample rate (must be integer multiple of symbol_rate).
//! - `bt`: Gaussian filter bandwidth-time product (0.5 for POCSAG).

use futuresdr::anyhow::Result;
use futuresdr::num_complex::Complex32;
use futuresdr::runtime::Block;
use futuresdr::runtime::BlockMeta;
use futuresdr::runtime::BlockMetaBuilder;
use futuresdr::runtime::Kernel;
use futuresdr::runtime::MessageIo;
use futuresdr::runtime::MessageIoBuilder;
use futuresdr::runtime::StreamIo;
use futuresdr::runtime::StreamIoBuilder;
use futuresdr::runtime::WorkIo;
use std::f32::consts::PI;

pub struct GaussFsk {
    deviation_hz: f32,
    symbol_rate: f32,
    sample_rate: f32,
    samples_per_symbol: usize,
    filter_taps: Vec<f32>,
    filter_state: Vec<f32>, // ring buffer of input symbols for convolution
    filter_idx: usize,
    phase: f32, // accumulated phase (radians)
}

impl GaussFsk {
    pub fn new(deviation_hz: f32, symbol_rate: f32, sample_rate: f32, bt: f32) -> Block {
        let samples_per_symbol = (sample_rate / symbol_rate).round() as usize;
        let filter_taps = gaussian_taps(bt, samples_per_symbol, 3); // span 3 symbols
        let filter_len = filter_taps.len();

        let kernel = Self {
            deviation_hz,
            symbol_rate,
            sample_rate,
            samples_per_symbol,
            filter_taps,
            filter_state: vec![0.0; filter_len],
            filter_idx: 0,
            phase: 0.0,
        };

        Block::new(
            BlockMetaBuilder::new("GaussFsk").build(),
            StreamIoBuilder::new()
                .add_input::<u8>("in") // 0 or 1 as u8
                .add_output::<Complex32>("out")
                .build(),
            MessageIoBuilder::new().build(),
            kernel,
        )
    }
}

#[async_trait::async_trait]
impl Kernel for GaussFsk {
    async fn work(
        &mut self,
        io: &mut WorkIo,
        sio: &mut StreamIo,
        _mio: &mut MessageIo<Self>,
        _meta: &mut BlockMeta,
    ) -> Result<()> {
        let input = sio.input(0).slice::<u8>();
        let output = sio.output(0).slice::<Complex32>();

        let n_symbols = input.len();
        let n_out_avail = output.len() / self.samples_per_symbol;
        let n = n_symbols.min(n_out_avail);

        if n == 0 {
            if sio.input(0).finished() {
                io.finished = true;
            }
            return Ok(());
        }

        let mut out_idx = 0;
        for sym_idx in 0..n {
            // Convert bool-as-u8 to ±1.0
            let sym_val = if input[sym_idx] != 0 { 1.0f32 } else { -1.0f32 };

            // For each output sample in this symbol period:
            for _s in 0..self.samples_per_symbol {
                // Push sample into filter state
                self.filter_state[self.filter_idx] = sym_val;
                self.filter_idx = (self.filter_idx + 1) % self.filter_state.len();

                // Convolve: compute filtered instantaneous frequency
                let mut freq = 0.0f32;
                for (tap_idx, &tap) in self.filter_taps.iter().enumerate() {
                    let state_idx =
                        (self.filter_idx + tap_idx) % self.filter_state.len();
                    freq += self.filter_state[state_idx] * tap;
                }

                // Frequency → phase increment
                let phase_inc = 2.0 * PI * self.deviation_hz * freq / self.sample_rate;
                self.phase += phase_inc;

                // Keep phase in [-pi, pi] to avoid precision loss
                if self.phase > PI {
                    self.phase -= 2.0 * PI;
                } else if self.phase < -PI {
                    self.phase += 2.0 * PI;
                }

                output[out_idx] = Complex32::new(self.phase.cos(), self.phase.sin());
                out_idx += 1;
            }
        }

        sio.input(0).consume(n);
        sio.output(0).produce(out_idx);

        if sio.input(0).finished() && n == n_symbols {
            io.finished = true;
        }

        Ok(())
    }
}

/// Generate Gaussian filter taps.
///
/// - `bt`: bandwidth-time product
/// - `samples_per_symbol`: oversampling factor
/// - `span_symbols`: filter spans this many symbol periods
fn gaussian_taps(bt: f32, samples_per_symbol: usize, span_symbols: usize) -> Vec<f32> {
    let n = span_symbols * samples_per_symbol;
    let mut taps = Vec::with_capacity(n);
    let sigma = (2.0 * PI * bt).recip() * (2.0f32.ln()).sqrt();
    let center = (n as f32 - 1.0) / 2.0;

    for i in 0..n {
        let t = (i as f32 - center) / samples_per_symbol as f32;
        let val = (-0.5 * (t / sigma).powi(2)).exp();
        taps.push(val);
    }

    // Normalize so taps sum to 1.0
    let sum: f32 = taps.iter().sum();
    for tap in &mut taps {
        *tap /= sum;
    }

    taps
}
```

```rust
    fn start(&self, params: serde_json::Value) -> Result<RunningApp> {

        let join = tokio::task::spawn_blocking(move || {

                PocsagMessageType::Alphanumeric => {

            let sample_rate = 2_400_000.0f32;

            let symbol_rate = params.baud_rate as f32;
            let bt = 0.5f32;

            // Build flowgraph
            let mut fg = Flowgraph::new();

            let src = VectorSource::<u8>::new(symbols);
            let modulator = GaussFsk::new(deviation_hz, symbol_rate, sample_rate, bt);
            let sink_cfg = HackRfSinkConfig {
                center_hz: params.center_hz,
                sample_rate: sample_rate as f64,
                vga_gain_db: params.vga_gain_db,
                amp_enabled: params.amp_enabled,
            };

                        message: e.to_string(),
                    });
                }
            }
        });

            stop: stop_tx,

        })
    }
```

```rust
        //   NFM (RX) → stop → POCSAG (TX) → stop → NFM (RX) → ...

        }
    }
```

```rust
//! Round-trip test: encode a POCSAG message, then decode the bitstream and verify
//! the original message is recovered. This validates the same invariant as
//! "decode with multimon-ng" but without requiring hardware or external tools.

use mayhem_protocols::pocsag::bch::encode_bch;
use mayhem_protocols::pocsag::codeword::sync_codeword;
use mayhem_protocols::pocsag::encoder::{encode_pocsag, MessageType, PocsagMessage};

/// Minimal POCSAG decoder: extracts address + message bits from a bitstream.
fn decode_bitstream(bits: &[bool]) -> Option<(u32, Vec<u8>)> {
    // Skip 576-bit preamble
    let bits = &bits[576..];
    
    // Find sync
    let sync_pattern = u32_to_bits(sync_codeword());
    if bits[..32] != sync_pattern[..] {
        return None;
    }
    let codewords = &bits[32..];
    
    // Parse first non-idle codeword as address
    let mut ric: u32 = 0;
    let mut msg_bits: Vec<bool> = Vec::new();
    let mut found_address = false;
    let mut frame_pos = 0;

    for (i, chunk) in codewords.chunks(32).enumerate() {
        if chunk.len() < 32 {
            break;
        }
        let cw = bits_to_u32(chunk);
        
        if cw == 0x7CD215D8 {
            // sync — new batch, reset frame counter
            frame_pos = 0;
            continue;
        }
        if cw == 0x7A89C197 {
            // idle
            frame_pos = i; // track position
            continue;
        }
        
        let flag = cw >> 31;
        if flag == 0 && !found_address {
            // Address codeword
            let addr_field = (cw >> 13) & 0x3FFFF;
            let func = (cw >> 11) & 0x03;
            // frame position from slot index
            let fp = (i % 16) / 2;
            ric = (addr_field << 3) | (fp as u32);
            found_address = true;
        } else if flag == 1 && found_address {
            // Message codeword — extract 20 data bits
            let data = (cw >> 11) & 0xFFFFF;
            for bit_idx in (0..20).rev() {
                msg_bits.push((data >> bit_idx) & 1 == 1);
            }
        }
    }

    if !found_address {
        return None;
    }

    // Convert message bits back to 7-bit ASCII (alphanumeric)
    let mut chars = Vec::new();
    for chunk in msg_bits.chunks(7) {
        if chunk.len() < 7 {
            break;
        }
        let mut val = 0u8;
        for (i, &b) in chunk.iter().enumerate() {
            if b {
                val |= 1 << i; // LSB first
            }
        }
        if val == 0 {
            break; // null terminator or padding
        }
        chars.push(val);
    }

    Some((ric, chars))
}

fn u32_to_bits(val: u32) -> Vec<bool> {
    (0..32).rev().map(|i| (val >> i) & 1 == 1).collect()
}

fn bits_to_u32(bits: &[bool]) -> u32 {
    let mut val = 0u32;
    for (i, &b) in bits.iter().enumerate() {
        if b {
            val |= 1 << (31 - i);
        }
    }
    val
}

#[test]
fn roundtrip_alphanumeric_512() {
    roundtrip_at_baud("HELLO WORLD", 1234567, 512);
}

#[test]
fn roundtrip_alphanumeric_1200() {
    roundtrip_at_baud("TEST MSG 1200", 42, 1200);
}

#[test]
fn roundtrip_alphanumeric_2400() {
    roundtrip_at_baud("QUICK TEST", 9999, 2400);
}

fn roundtrip_at_baud(text: &str, ric: u32, baud: u16) {
    let msg = PocsagMessage {
        ric,
        function: 0,
        content: MessageType::Alphanumeric(text.to_string()),
        baud_rate: baud,
    };
    let bits = encode_pocsag(&msg);
    
    let (decoded_ric, decoded_chars) = decode_bitstream(&bits)
        .expect("failed to decode bitstream");
    
    assert_eq!(decoded_ric, ric, "RIC mismatch at {baud} baud");
    let decoded_text: String = decoded_chars.iter().map(|&c| c as char).collect();
    assert!(
        decoded_text.starts_with(text),
        "Message mismatch at {baud} baud: expected '{text}', got '{decoded_text}'"
    );
}
```
